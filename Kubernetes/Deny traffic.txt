kind: NetworkPolicy
apiVersion: networking.k8s.io/v1
metadata:
  name: deny-all-outgoing
spec:
  podSelector: {}
  policyTypes:
  - Egress
  egress: []
---
kind: NetworkPolicy
apiVersion: networking.k8s.io/v1
metadata:
  name: block-frontend-outgoing
spec:
  podSelector:
    matchLabels:
      app: frontend
  policyTypes:
  - Egress
  egress:
  - to:
    - podSelector:
        matchLabels:
          app: backend
---
kind: NetworkPolicy
apiVersion: networking.k8s.io/v1
metadata:
  name: allow-frontend-to-database-backend
spec:
  podSelector:
    matchLabels:
      app: frontend
  policyTypes:
  - Egress
  egress:
  - to:
    - podSelector:
        matchLabels:
          app: database
    - podSelector:
        matchLabels:
          app: backend

---
